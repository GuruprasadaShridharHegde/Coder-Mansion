#include<stdio.h>
using namespace std;
int main()
{
  cout<<"Enter a number: "; int check; cin>>check;
  //checking whether the number is even or odd
  if(check % 2 == 0)

  }
  else
  {
     cout<<check<<"is an odd number";
  }
  return 0;
}
